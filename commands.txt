
git status //чекнуть статус
git pull//подтянуть файлы с гита и обновить свои файлы до их версии
clear //очистить экран

***РАБОТА С ВЕТКАМИ***

git branch //чекнуть ветки
git branch name//создать ветку нейм
git branch -D name//удалить ветку нейм
git checkout name//перейти на ветку name
gitk --all&//перейти в список всех бранчей

***РАБОТА С ФАЙЛАМИ***

git add .//добавить все файлы в гит
git rm --cached name//удалить name
git commit -m "hello git"//добавить коммит с названием hello git

git push// запушить на гитхаб
